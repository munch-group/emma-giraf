IM model med ghost population i baboon data

I denne notebook analyserer jeg genetiske data fra bavianer (Papio).

Jeg bruger:

  • sgkit til at håndtere genomdata
  • phasic til likelihood-baseret inference
  • en Isolation-with-Migration (IM) model

IM (Isolation-with-Migration) modellen beskriver:

  • To populationer, der splitter fra en fælles forfader
  • Efter split kan der være migration mellem dem

Jeg udvider modellen med en “ghost population”:

  • En usamplet population som bidrager genetisk
  • Dette er relevant for bavianer, hvor studier viser kompleks admixture

import msprime
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
%config InlineBackend.figure_format = 'retina'

def derived_counts(ts, rec_rate):
    records = []
    for var in ts.variants():
        p, g = var.site.position, var.genotypes
        records.append((int(p), p*rec_rate, g.sum()))
    df = pd.DataFrame().from_records(
        records, columns=["pos", "gen_pos", "count"]
        )
    return df

mut_rate = 5e-10
rec_rate = 1e-8
nr_samples = 5
seq_length = 100_000_000
pop1_size, pop2_size, anc_pop_size = 20_000, 10_000, 15_000
migr_pop1_to_pop2 = 1e-4
migr_pop2_to_pop1 = 5e-4

demography = msprime.Demography()
demography.add_population(name="pop1", initial_size=pop1_size)
demography.add_population(name="pop2", initial_size=pop2_size)
demography.set_migration_rate(source="pop1", dest="pop2", rate=migr_pop1_to_pop2)
demography.set_migration_rate(source="pop2", dest="pop1", rate=migr_pop2_to_pop1)
demography.add_population(name="ancestral", initial_size=anc_pop_size)
demography.add_population_split(time=1000, derived=["pop1", "pop2"], ancestral="ancestral")
ts = msprime.sim_ancestry(samples={"pop1": nr_samples, "pop2": 0}, ploidy=1, 
                          demography=demography, recombination_rate=rec_rate, 
                          sequence_length=seq_length, random_seed=12)
ts = msprime.sim_mutations(ts, rate=mut_rate, random_seed=5678)
df = derived_counts(ts, rec_rate)
df.to_csv("IM_model_derived_counts.csv", index=False)
def pairs_in_range(nums, diff_lo, diff_hi):
    n = len(nums)
    lo, hi = 1, 1
    pairs = []
    for i in range(n):
        if lo <= i:
            lo = i + 1
        while lo < n and nums[lo] - nums[i] < diff_lo:
            lo += 1
        if hi <= i:
            hi = i + 1
        while hi < n and nums[hi] - nums[i] <= diff_hi:
            hi += 1
        for j in range(lo, hi):
            pairs.append((i, j))
    return pairs

df = pd.read_csv("IM_model_derived_counts.csv")

col = "pos" # can also use "gen_pos"
distance, tolerance = 5000, 500
min_dist, max_dist = distance - tolerance, distance + tolerance
records = []
for i, j in pairs_in_range(df[col].values, min_dist, max_dist):
    records.append((df.at[i, col], df.at[j, col], df.at[i, "count"], df.at[j, "count"]))
pairs = pd.DataFrame.from_records(records, columns=["pos1", "pos2", "count1", "count2"])

      IM model (uden ghost)

      IM model:

      ancestral | split /
      pop1 pop2

      Efter split kan der være migration mellem populationer.

      graph = Graph()

graph.add_population("pop1")
graph.add_population("pop2")
graph.add_population("ancestral")

graph.add_split(
    parent="ancestral",
    children=["pop1", "pop2"],
    time="T_split"
)

graph.add_migration("pop1", "pop2", rate="m12")
graph.add_migration("pop2", "pop1", rate="m21")
      priors = PropertySet(
    T_split=GaussPrior(1000, 500),
    m12=HalfCauchyPrior(1e-4),
    m21=HalfCauchyPrior(1e-4),
)
      result = graph.fit(
    observations=observations,
    priors=priors,
    optimizer=Adamelia(
        step_size=ExpStepSize(0.05),
        regularization=ExpRegularization(1e-3)
    ),
    steps=500
)

print(result)

      Ghost model

      Ghost model: ancestral /
      ghost pop2 |. | pop 1 pop1 modtager admixture fra både pop2 og en ukendt (ghost) population.

      graph_ghost = Graph()

graph_ghost.add_population("pop1")
graph_ghost.add_population("pop2")
graph_ghost.add_population("ghost")
graph_ghost.add_population("ancestral")

graph_ghost.add_split(
    parent="ancestral",
    children=["ghost", "pop2"],
    time="T_anc"
)

graph_ghost.add_admixture(
    child="pop1",
    parents=["pop2", "ghost"],
    proportions=["p", "1-p"],
    time="T_admix"
)
      priors_ghost = PropertySet(
    T_anc=GaussPrior(1500, 500),
    T_admix=GaussPrior(500, 200),
    p=GaussPrior(0.7, 0.2),
)
      result_ghost = graph_ghost.fit(
    observations=observations,
    priors=priors_ghost,
    optimizer=Adamelia(
